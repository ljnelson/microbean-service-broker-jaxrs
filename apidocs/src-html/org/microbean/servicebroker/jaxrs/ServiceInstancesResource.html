073  @GET
074  @Path("{instance_id}/last_operation")
075  @Consumes(MediaType.APPLICATION_JSON)
076  public Response getLastOperation(@PathParam("instance_id") final String instanceId,
077                                   @QueryParam("service_id") final String serviceId,
078                                   @QueryParam("plan_id") final String planId,
079                                   @QueryParam("operation") final String operationId)
080    throws ServiceBrokerException {
081    final String cn = this.getClass().getName();
082    final String mn = "getLastOperation";
083    final Logger logger = Logger.getLogger(cn);
084    assert logger != null;
085    if (logger.isLoggable(Level.FINER)) {
086      logger.entering(cn, mn, new Object[] { instanceId, serviceId, planId, operationId });
087    }
088    Objects.requireNonNull(instanceId);
089
090    // The specification implies but does not state that operation is required.
091    // https://github.com/openservicebrokerapi/servicebroker/blob/v2.13/spec.md#parameters
092    if (operationId == null) {
093      throw new BadRequestException("The operation query parameter was not specified");
094    }
095    
096    final Response returnValue;
097    Response temp = null;
098    final LastOperationQuery query = new LastOperationQuery(serviceId, instanceId, planId, operationId);
099    try {
100      final LastOperation lastOperation = this.serviceBroker.getLastOperation(query);
101      if (lastOperation == null) {
102        temp = Response.serverError().entity("{}").build();
103      } else {
104        temp = Response.ok().entity(lastOperation).build();
105      }
106    } catch (final NoSuchServiceInstanceException noSuchServiceInstanceException) {
107      // Normally exceptions are handled by appropriate mappers.  This
108      // is a special case.  See
109      // https://github.com/openservicebrokerapi/servicebroker/blob/v2.13/spec.md#response-1
110      temp = Response.status(410).entity("{}").build();
111    } finally {
112      returnValue = temp;
113    }    
114    
115    if (logger.isLoggable(Level.FINER)) {
116      logger.exiting(cn, mn, returnValue);
117    }
118    return returnValue;
119  }

121  @PUT
122  @Path("{instance_id}")
123  @Consumes(MediaType.APPLICATION_JSON)
124  public Response putServiceInstance(@PathParam("instance_id") final String instanceId,
125                                     final ProvisionServiceInstanceCommand command,
126                                     @QueryParam("accepts_incomplete") final boolean acceptsIncomplete)
127    throws ServiceBrokerException {
128    final String cn = this.getClass().getName();
129    final String mn = "putServiceInstance";
130    final Logger logger = Logger.getLogger(cn);
131    assert logger != null;
132    if (logger.isLoggable(Level.FINER)) {
133      logger.entering(cn, mn, new Object[] { instanceId, command });
134    }
135    Objects.requireNonNull(instanceId);
136    Objects.requireNonNull(command);
137
138    command.setAcceptsIncomplete(acceptsIncomplete);
139    
140    if (command.getInstanceId() == null) {
141      command.setInstanceId(instanceId);
142    }
143
144    final Response returnValue;
145    if (!acceptsIncomplete && this.serviceBroker.isAsynchronousOnly()) {
146      // See https://github.com/openservicebrokerapi/servicebroker/blob/v2.13/spec.md#asynchronous-operations
147      returnValue = Response.status(422).entity("{\n" +
148                                                "  \"error\": \"AsyncRequired\",\n" +
149                                                "  \"description\": \"This service plan requires client support for asynchronous service operations.\"\n" +
150                                                "}").build();
151    } else {
152      final ProvisionServiceInstanceCommand.Response commandResponse = this.serviceBroker.execute(command);
153      if (commandResponse == null) {
154        returnValue = Response.serverError().entity("{}").build();
155      } else if (commandResponse.getOperation() == null) {        
156        // The specification mandates a 201 return code, but does not
157        // say what the Location: header should contain, so we don't
158        // return one.
159        // See https://github.com/openservicebrokerapi/servicebroker/blob/v2.13/spec.md#response-2
160        returnValue = Response.status(201).entity(commandResponse).build();
161      } else {
162        // The command response contained an operation property, so
163        // that means it's asynchronous.
164        // See https://github.com/openservicebrokerapi/servicebroker/blob/v2.13/spec.md#response-2
165        returnValue = Response.status(202).entity(commandResponse).build();
166      }
167    }
168
169    if (logger.isLoggable(Level.FINER)) {
170      logger.exiting(cn, mn, returnValue);
171    }
172    return returnValue;
173  }
